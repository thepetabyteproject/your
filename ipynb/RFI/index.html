def plot_data(data, mask=None):
    plt.figure(figsize=(10, 8))
    if mask is not None:
        data[:, mask] = 0
    plt.imshow(data.T, aspect="auto")
    if mask is not None:
        for line in np.arange(data.shape[1])[mask]:
            plt.axhline(line, color="r", xmin=0, xmax=0.025, lw=0.1)

    plt.xlabel("Time Samples")
    plt.ylabel("Frequency Channels")
    plt.colorbar()
    return plt.show()
